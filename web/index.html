Install in 5 minutes

Pick your surface. Same brain, same data underneath.

git clone https://github.com/adityaarsharma/perch
cd perch && npm install && npm run build

# Add to ~/.claude/claude_desktop_config.json
{
  "mcpServers": {
    "perch": {
      "command": "node",
      "args": ["/absolute/path/to/perch/dist/index.js"],
      "env": { "RUNCLOUD_API_KEY": "your-runcloud-api-key" }
    }
  }
}

# Restart Claude Code · type /perch_brain to verify
ssh root@your-server
git clone https://github.com/adityaarsharma/perch /opt/perch
cd /opt/perch/telegram-bot
cp config.example.env .env
# Edit .env — bot token, chat ID, fix-server token
./setup.sh
# Telegram bot live in < 5 min
# Claude Code MCP for deep work
git clone https://github.com/adityaarsharma/perch
cd perch && npm install && npm run build

# Telegram bot for 4am pages — same repo, on your VPS
ssh root@your-server
git clone https://github.com/adityaarsharma/perch /opt/perch
cd /opt/perch/telegram-bot && cp config.example.env .env && ./setup.sh

# Same brain, same data, two surfaces.
